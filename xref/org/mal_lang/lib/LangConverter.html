55  /**
56   * Class for converting {@link org.mal_lang.lib.AST} objects into {@link org.mal_lang.langspec.Lang}
57   * objects.
58   *
59   * @since 0.1.0
60   */
61  public class LangConverter {
62    private final MalLogger LOGGER;
63    private final Map<String, List<AST.Category>> astCategories = new LinkedHashMap<>();
64    private final List<AST.Association> astAssociations = new ArrayList<>();
65    private final Map<String, String> astDefines = new LinkedHashMap<>();
66  
67    private LangConverter(AST ast, boolean verbose, boolean debug) {
68      Locale.setDefault(Locale.ROOT);
69      this.LOGGER = new MalLogger("LANG_CONVERTER", verbose, debug);
70  
71      // Collect defines
72      for (var astDefine : ast.getDefines()) {
73        this.astDefines.put(astDefine.key.id, astDefine.value);
74      }
75  
76      // Collect categories
77      var allAstCategories = ast.getCategories();
78      for (var astCategory : allAstCategories) {
79        if (!this.astCategories.containsKey(astCategory.name.id)) {
80          this.astCategories.put(astCategory.name.id, new ArrayList<>());
81        }
82        this.astCategories.get(astCategory.name.id).add(astCategory);
83      }
84  
85      // Collect associations
86      for (var astAssociation : ast.getAssociations()) {
87        this.astAssociations.add(astAssociation);
88      }
89    }
90  
91    /**
92     * Converts an {@link org.mal_lang.lib.AST} object into a {@link org.mal_lang.langspec.Lang}
93     * object.
94     *
95     * @param ast the {@link org.mal_lang.lib.AST} to convert
96     * @return a {@link org.mal_lang.langspec.Lang}
97     * @since 0.1.0
98     */
99    public static Lang convert(AST ast) {
100     return convert(ast, false, false);
101   }
102 
103   /**
104    * Converts an {@link org.mal_lang.lib.AST} object into a {@link org.mal_lang.langspec.Lang}
105    * object.
106    *
107    * @param ast the {@link org.mal_lang.lib.AST} to convert
108    * @param verbose whether verbose information should be logged
109    * @param debug whether debug information should be logged
110    * @return a {@link org.mal_lang.langspec.Lang}
111    * @since 0.1.0
112    */
113   public static Lang convert(AST ast, boolean verbose, boolean debug) {
114     return new LangConverter(ast, verbose, debug).convertLog();
115   }
116 
117   private Lang convertLog() {
118     var lang = this.convertLang();
119     LOGGER.print();
120     return lang;
121   }

123   private Lang convertLang() {
124     var langBuilder = new LangBuilder();
125 
126     for (var entry : this.astDefines.entrySet()) {
127       langBuilder.addDefine(entry.getKey(), entry.getValue());
128     }
129 
130     for (var entry : this.astCategories.entrySet()) {
131       var categoryBuilder = new CategoryBuilder(entry.getKey());
132       for (var astCategory : entry.getValue()) {
133         for (var meta : astCategory.meta) {
134           categoryBuilder.getMeta().addEntry(meta.type.id, meta.string);
135         }
136       }
137       langBuilder.addCategory(categoryBuilder);
138     }
139 
140     for (var astCategories : this.astCategories.values()) {
141       for (var astCategory : astCategories) {
142         for (var astAsset : astCategory.assets) {
143           var assetBuilder =
144               new AssetBuilder(
145                   astAsset.name.id,
146                   astCategory.name.id,
147                   astAsset.isAbstract,
148                   astAsset.parent.isPresent() ? astAsset.parent.get().id : null);
149           for (var meta : astAsset.meta) {
150             assetBuilder.getMeta().addEntry(meta.type.id, meta.string);
151           }
152           for (var astVariable : astAsset.variables) {
153             var variableBuilder =
154                 new VariableBuilder(
155                     astVariable.name.id, this.convertStepExpression(astVariable.expr, false));
156             assetBuilder.addVariable(variableBuilder);
157           }
158           for (var astAttackStep : astAsset.attackSteps) {
159             var name = astAttackStep.name.id;
160             var type = this.convertAttackStepType(astAttackStep.type);
161             var risk =
162                 astAttackStep.cia.isPresent() ? this.convertRisk(astAttackStep.cia.get()) : null;
163             var ttc =
164                 astAttackStep.ttc.isPresent()
165                     ? this.convertTtcExpression(astAttackStep.ttc.get())
166                     : null;
167             var requires =
168                 astAttackStep.requires.isPresent()
169                     ? this.convertRequires(astAttackStep.requires.get())
170                     : null;
171             var reaches =
172                 astAttackStep.reaches.isPresent()
173                     ? this.convertReaches(astAttackStep.reaches.get())
174                     : null;
175             var attackStepBuilder = new AttackStepBuilder(name, type, risk, ttc, requires, reaches);
176             for (var meta : astAttackStep.meta) {
177               attackStepBuilder.getMeta().addEntry(meta.type.id, meta.string);
178             }
179             for (var tag : astAttackStep.tags) {
180               attackStepBuilder.addTag(tag.id);
181             }
182             assetBuilder.addAttackStep(attackStepBuilder);
183           }
184           langBuilder.addAsset(assetBuilder);
185         }
186       }
187     }
188 
189     for (var astAssociation : this.astAssociations) {
190       var associationBuilder =
191           new AssociationBuilder(
192               astAssociation.linkName.id,
193               astAssociation.leftAsset.id,
194               astAssociation.leftField.id,
195               this.convertMultiplicity(astAssociation.leftMult),
196               astAssociation.rightAsset.id,
197               astAssociation.rightField.id,
198               this.convertMultiplicity(astAssociation.rightMult));
199       for (var meta : astAssociation.meta) {
200         associationBuilder.getMeta().addEntry(meta.type.id, meta.string);
201       }
202       langBuilder.addAssociation(associationBuilder);
203     }
204 
205     return Lang.fromBuilder(langBuilder);
206   }

208   private AttackStepType convertAttackStepType(AST.AttackStepType astType) {
209     switch (astType) {
210       case ANY:
211         return AttackStepType.OR;
212       case ALL:
213         return AttackStepType.AND;
214       case DEFENSE:
215         return AttackStepType.DEFENSE;
216       case EXIST:
217         return AttackStepType.EXIST;
218       case NOTEXIST:
219         return AttackStepType.NOT_EXIST;
220       default:
221         throw new RuntimeException(String.format("Invalid attack step type %s", astType));
222     }
223   }

225   private Risk convertRisk(List<AST.CIA> astCiaList) {
226     boolean isConfidentiality = false;
227     boolean isIntegrity = false;
228     boolean isAvailability = false;
229     for (var astCia : astCiaList) {
230       switch (astCia) {
231         case C:
232           isConfidentiality = true;
233           break;
234         case I:
235           isIntegrity = true;
236           break;
237         case A:
238           isAvailability = true;
239           break;
240       }
241     }
242     return new Risk(isConfidentiality, isIntegrity, isAvailability);
243   }

245   private TtcExpression convertTtcExpression(AST.TTCExpr astTtcExpression) {
246     if (astTtcExpression instanceof AST.TTCAddExpr) {
247       var astTtcAddition = (AST.TTCAddExpr) astTtcExpression;
248       return new TtcAddition(
249           this.convertTtcExpression(astTtcAddition.lhs),
250           this.convertTtcExpression(astTtcAddition.rhs));
251     } else if (astTtcExpression instanceof AST.TTCSubExpr) {
252       var astTtcSubtraction = (AST.TTCSubExpr) astTtcExpression;
253       return new TtcSubtraction(
254           this.convertTtcExpression(astTtcSubtraction.lhs),
255           this.convertTtcExpression(astTtcSubtraction.rhs));
256     } else if (astTtcExpression instanceof AST.TTCMulExpr) {
257       var astTtcMultiplication = (AST.TTCMulExpr) astTtcExpression;
258       return new TtcMultiplication(
259           this.convertTtcExpression(astTtcMultiplication.lhs),
260           this.convertTtcExpression(astTtcMultiplication.rhs));
261     } else if (astTtcExpression instanceof AST.TTCDivExpr) {
262       var astTtcDivision = (AST.TTCDivExpr) astTtcExpression;
263       return new TtcDivision(
264           this.convertTtcExpression(astTtcDivision.lhs),
265           this.convertTtcExpression(astTtcDivision.rhs));
266     } else if (astTtcExpression instanceof AST.TTCPowExpr) {
267       var astTtcExponentiation = (AST.TTCPowExpr) astTtcExpression;
268       return new TtcExponentiation(
269           this.convertTtcExpression(astTtcExponentiation.lhs),
270           this.convertTtcExpression(astTtcExponentiation.rhs));
271     } else if (astTtcExpression instanceof AST.TTCFuncExpr) {
272       var astTtcFunction = (AST.TTCFuncExpr) astTtcExpression;
273       return new TtcFunction(
274           TtcDistribution.fromString(astTtcFunction.name.id),
275           astTtcFunction.params.stream().mapToDouble(x -> x).toArray());
276     } else if (astTtcExpression instanceof AST.TTCNumExpr) {
277       var astTtcNumber = (AST.TTCNumExpr) astTtcExpression;
278       return new TtcNumber(astTtcNumber.value);
279     } else {
280       throw new RuntimeException(
281           String.format("Invalid TTC expression type %s", astTtcExpression.getClass().getName()));
282     }
283   }

285   private StepsBuilder convertRequires(AST.Requires astRequires) {
286     var stepsBuilder = new StepsBuilder(true);
287     for (var astStepExpression : astRequires.requires) {
288       stepsBuilder.addStepExpression(this.convertStepExpression(astStepExpression, false));
289     }
290     return stepsBuilder;
291   }
292 
293   private StepsBuilder convertReaches(AST.Reaches astReaches) {
294     var stepsBuilder = new StepsBuilder(!astReaches.inherits);
295     for (var astStepExpression : astReaches.reaches) {
296       stepsBuilder.addStepExpression(this.convertStepExpression(astStepExpression, true));
297     }
298     return stepsBuilder;
299   }

301   private StepExpressionBuilder convertStepExpression(AST.Expr astExpr, boolean isAttackStep) {
302     if (astExpr instanceof AST.UnionExpr) {
303       var astStepUnion = (AST.UnionExpr) astExpr;
304       return new StepUnionBuilder(
305           this.convertStepExpression(astStepUnion.lhs, false),
306           this.convertStepExpression(astStepUnion.rhs, isAttackStep));
307     } else if (astExpr instanceof AST.IntersectionExpr) {
308       var astStepIntersection = (AST.IntersectionExpr) astExpr;
309       return new StepIntersectionBuilder(
310           this.convertStepExpression(astStepIntersection.lhs, false),
311           this.convertStepExpression(astStepIntersection.rhs, isAttackStep));
312     } else if (astExpr instanceof AST.DifferenceExpr) {
313       var astStepDifference = (AST.DifferenceExpr) astExpr;
314       return new StepDifferenceBuilder(
315           this.convertStepExpression(astStepDifference.lhs, false),
316           this.convertStepExpression(astStepDifference.rhs, isAttackStep));
317     } else if (astExpr instanceof AST.StepExpr) {
318       var astStepCollect = (AST.StepExpr) astExpr;
319       return new StepCollectBuilder(
320           this.convertStepExpression(astStepCollect.lhs, false),
321           this.convertStepExpression(astStepCollect.rhs, isAttackStep));
322     } else if (astExpr instanceof AST.TransitiveExpr) {
323       var astStepTransitive = (AST.TransitiveExpr) astExpr;
324       return new StepTransitiveBuilder(this.convertStepExpression(astStepTransitive.e, false));
325     } else if (astExpr instanceof AST.SubTypeExpr) {
326       var astStepSubType = (AST.SubTypeExpr) astExpr;
327       return new StepSubTypeBuilder(
328           astStepSubType.subType.id, this.convertStepExpression(astStepSubType.e, false));
329     } else if (astExpr instanceof AST.IDExpr) {
330       var astStepId = (AST.IDExpr) astExpr;
331       if (isAttackStep) {
332         return new StepAttackStepBuilder(astStepId.id.id);
333       } else {
334         return new StepFieldBuilder(astStepId.id.id);
335       }
336     } else if (astExpr instanceof AST.CallExpr) {
337       var astStepVariable = (AST.CallExpr) astExpr;
338       return new StepVariableBuilder(astStepVariable.id.id);
339     } else {
340       throw new RuntimeException(
341           String.format("Invalid step expression type %s", astExpr.getClass().getName()));
342     }
343   }

345   private Multiplicity convertMultiplicity(AST.Multiplicity astMultiplicity) {
346     switch (astMultiplicity) {
347       case ZERO_OR_ONE:
348         return Multiplicity.ZERO_OR_ONE;
349       case ZERO_OR_MORE:
350         return Multiplicity.ZERO_OR_MORE;
351       case ONE:
352         return Multiplicity.ONE;
353       case ONE_OR_MORE:
354         return Multiplicity.ONE_OR_MORE;
355       default:
356         throw new RuntimeException(String.format("Invalid multiplicity %s", astMultiplicity));
357     }
358   }
