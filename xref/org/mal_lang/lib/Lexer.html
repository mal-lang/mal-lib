64      escapeSequences.put("\\b", (byte) '\b');
65      escapeSequences.put("\\n", (byte) '\n');
66      escapeSequences.put("\\t", (byte) '\t');
67      escapeSequences.put("\\r", (byte) '\r');

85    public Lexer(File file, String relativeName, boolean verbose, boolean debug) throws IOException {
86      Locale.setDefault(Locale.ROOT);
87      LOGGER = new MalLogger("LEXER", verbose, debug);
88      try {
89        LOGGER.debug(String.format("Creating lexer with file '%s'", relativeName));
90        if (!file.exists()) {
91          throw new IOException(String.format("%s: No such file or directory", relativeName));
92        }
93        this.filename = relativeName;
94        this.input = Files.readAllBytes(file.toPath());
95        this.index = 0;
96        this.line = 1;
97        this.col = 1;
98        this.eof = input.length == 0;
99      } catch (IOException e) {
100       LOGGER.print();
101       throw e;
102     }
103   }

105   public static boolean syntacticallyEqual(Lexer l1, Lexer l2) {
106     try {
107       var tok1 = l1.next();
108       var tok2 = l2.next();
109       while (tok1.type != TokenType.EOF && tok2.type != TokenType.EOF) {
110         if (tok1.type != tok2.type
111             || !tok1.stringValue.equals(tok2.stringValue)
112             || tok1.intValue != tok2.intValue
113             || tok1.doubleValue != tok2.doubleValue) {
114           return false;
115         }
116         tok1 = l1.next();
117         tok2 = l2.next();
118       }
119       return tok1.type == TokenType.EOF && tok2.type == TokenType.EOF;
120     } catch (CompilerException e) {
121       return false;
122     }
123   }

125   private String getLexemeString() {
126     byte[] byteArray = new byte[lexeme.size()];
127     for (int i = 0; i < lexeme.size(); i++) {
128       byteArray[i] = lexeme.get(i).byteValue();
129     }
130     return new String(byteArray, StandardCharsets.UTF_8);
131   }

133   public Token next() throws CompilerException {
134     startLine = line;
135     startCol = col;
136     lexeme = new ArrayList<>();
137     if (eof) {
138       LOGGER.print();
139       return createToken(TokenType.EOF);
140     }
141     byte c = consume();
142     switch (c) {
143       case ' ':
144       case '\t':
145       case '\r':
146       case '\n':
147         return next();
148       case '#':
149         return createToken(TokenType.HASH);
150       case ':':
151         return createToken(TokenType.COLON);
152       case '{':
153         return createToken(TokenType.LCURLY);
154       case '}':
155         return createToken(TokenType.RCURLY);
156       case '+':
157         if (peek('>')) {
158           consume();
159           return createToken(TokenType.INHERIT);
160         } else {
161           return createToken(TokenType.PLUS);
162         }
163       case '-':
164         if (peek('>')) {
165           consume();
166           return createToken(TokenType.OVERRIDE);
167         } else if (peek("->")) {
168           consume(2);
169           return createToken(TokenType.RARROW);
170         } else {
171           return createToken(TokenType.MINUS);
172         }
173       case '&':
174         return createToken(TokenType.ALL);
175       case '|':
176         return createToken(TokenType.ANY);
177       case '!':
178         if (peek('E')) {
179           consume();
180           return createToken(TokenType.NOTEXIST);
181         } else {
182           throw exception("Expected 'E'");
183         }
184       case '@':
185         return createToken(TokenType.AT);
186       case '[':
187         return createToken(TokenType.LBRACKET);
188       case ']':
189         return createToken(TokenType.RBRACKET);
190       case '(':
191         return createToken(TokenType.LPAREN);
192       case ')':
193         return createToken(TokenType.RPAREN);
194       case ',':
195         return createToken(TokenType.COMMA);
196       case '<':
197         if (peek("--")) {
198           consume(2);
199           return createToken(TokenType.LARROW);
200         } else if (peek('-')) {
201           consume();
202           return createToken(TokenType.REQUIRE);
203         } else {
204           throw exception("Expected '-' or '--'");
205         }
206       case '=':
207         return createToken(TokenType.ASSIGN);
208       case '\\':
209         if (peek('/')) {
210           consume();
211           return createToken(TokenType.UNION);
212         } else {
213           throw exception("Expected '/'");
214         }
215       case '/':
216         if (peek('\\')) {
217           consume();
218           return createToken(TokenType.INTERSECT);
219         } else if (peek('/')) {
220           while (!eof && !peek('\n') && !peek('\r')) {
221             consume();
222           }
223           createComment(TokenType.SINGLECOMMENT);
224           return next();
225         } else if (peek('*')) {
226           consume();
227           while (!peek("*/")) {
228             if (eof) {
229               throw exception(
230                   String.format(
231                       "Unterminated comment starting at %s",
232                       new Position(filename, startLine, startCol)));
233             }
234             consume();
235           }
236           consume(2);
237           createComment(TokenType.MULTICOMMENT);
238           return next();
239         } else {
240           return createToken(TokenType.DIVIDE);
241         }
242       case '.':
243         if (peek('.')) {
244           consume();
245           return createToken(TokenType.RANGE);
246         } else {
247           return createToken(TokenType.DOT);
248         }
249       case '*':
250         return createToken(TokenType.STAR);
251       case '^':
252         return createToken(TokenType.POWER);
253       case '"':
254         while (!peek('"')) {
255           if (peek('\\')) {
256             consume();
257             if (eof || peek('\n')) {
258               throw exception(
259                   String.format(
260                       "Unterminated string starting at %s",
261                       new Position(filename, startLine, startCol)));
262             }
263             if (input[index] < 32 || input[index] > 126) {
264               throw exception(String.format("Invalid escape byte 0x%02X", input[index]));
265             }
266             consume();
267             var lexemeString = getLexemeString();
268             String escapeSequence = lexemeString.substring(lexemeString.length() - 2);
269             lexeme = lexeme.subList(0, lexeme.size() - 2);
270             if (!escapeSequences.containsKey(escapeSequence)) {
271               throw exception(String.format("Invalid escape sequence '%s'", escapeSequence));
272             }
273             lexeme.add(escapeSequences.get(escapeSequence));
274           } else if (eof || peek('\n')) {
275             throw exception(
276                 String.format(
277                     "Unterminated string starting at %s",
278                     new Position(filename, startLine, startCol)));
279           } else {
280             consume();
281           }
282         }
283         consume();
284         return createToken(TokenType.STRING);
285       default:
286         if (isAlpha(c)) {
287           while (isAlphaNumeric()) {
288             consume();
289           }
290           var lexemeString = getLexemeString();
291           if (keywords.containsKey(lexemeString)) {
292             return createToken(keywords.get(lexemeString));
293           } else {
294             return createToken(TokenType.ID);
295           }
296         } else if (isDigit(c)) {
297           while (isDigit()) {
298             consume();
299           }
300           if (peek("..") || !peek('.')) {
301             return createToken(TokenType.INT);
302           } else if (peek('.')) {
303             consume();
304             while (isDigit()) {
305               consume();
306             }
307             return createToken(TokenType.FLOAT);
308           }
309         }
310         if (c < 0) {
311           throw exception(String.format("Unexpected token 0x%02X", c));
312         } else {
313           throw exception(String.format("Unexpected token '%c'", (char) c));
314         }
315     }
316   }

318   private void consume(int n) {
319     for (int i = 0; i < n; i++) {
320       consume();
321     }
322   }
323 
324   private byte consume() {
325     if (eof) {
326       throw new RuntimeException("Consuming past end-of-file");
327     }
328     if (input[index] == (byte) '\n') {
329       line++;
330       col = 1;
331     } else {
332       col++;
333     }
334     var c = input[index++];
335     lexeme.add(c);
336     if (index == input.length) {
337       eof = true;
338     }
339     return c;
340   }

342   private boolean peek(String s) {
343     var bytes = s.getBytes();
344     if (input.length - index < bytes.length) {
345       return false;
346     }
347     for (int i = 0; i < bytes.length; i++) {
348       if (bytes[i] != input[index + i]) {
349         return false;
350       }
351     }
352     return true;
353   }
354 
355   private boolean peek(char c) {
356     return peek((byte) c);
357   }
358 
359   private boolean peek(byte c) {
360     if (eof) {
361       return false;
362     } else {
363       return c == input[index];
364     }
365   }

367   private void createComment(TokenType type) {
368     var lexemeString = getLexemeString();
369     lexemeString = lexemeString.substring(2, lexemeString.length());
370     if (type == TokenType.MULTICOMMENT) {
371       lexemeString = lexemeString.substring(0, lexemeString.length() - 2);
372     }
373     comments.add(new Token(type, filename, startLine, startCol, lexemeString));
374   }

376   private Token createRawToken(TokenType type) {
377     switch (type) {
378       case INT:
379         return new Token(type, filename, startLine, startCol, Integer.parseInt(getLexemeString()));
380       case FLOAT:
381         return new Token(
382             type, filename, startLine, startCol, Double.parseDouble(getLexemeString()));
383       case ID:
384         return new Token(type, filename, startLine, startCol, getLexemeString());
385       case STRING:
386         var lexemeString = getLexemeString();
387         return new Token(
388             type,
389             filename,
390             startLine,
391             startCol,
392             lexemeString.substring(1, lexemeString.length() - 1));
393       default:
394         return new Token(type, filename, startLine, startCol);
395     }
396   }

398   @SuppressWarnings("fallthrough")
399   private void readTrailingComments() throws CompilerException {
400     // Trailing comments are all comments followed on the same line as the previous
401     // token, including comments that follow previous trailing comments by exactly 1
402     // line.
403     startLine = line;
404     startCol = col;
405     lexeme = new ArrayList<>();
406     if (eof || peek('\n')) {
407       return;
408     }
409     byte c = consume();
410     switch (c) {
411       case ' ':
412       case '\t':
413         readTrailingComments();
414         return;
415       case '/':
416         if (peek('/')) {
417           while (!eof && !peek('\n') && !peek('\r')) {
418             consume();
419           }
420           createComment(TokenType.SINGLECOMMENT);
421           if (peek("\r\n")) {
422             consume(2);
423             readTrailingComments();
424           } else if (peek('\n')) {
425             consume();
426             readTrailingComments();
427           }
428           return;
429         } else if (peek('*')) {
430           consume();
431           while (!peek("*/")) {
432             if (eof) {
433               throw exception(
434                   String.format(
435                       "Unterminated comment starting at %s",
436                       new Position(filename, startLine, startCol)));
437             }
438             consume();
439           }
440           consume(2);
441           createComment(TokenType.MULTICOMMENT);
442           readTrailingComments();
443           return;
444         }
445         // Not a comment, we want to fall-through
446       default:
447         index--;
448         col--;
449         eof = false;
450         return;
451     }
452   }
453 
454   private Token createToken(TokenType type) throws CompilerException {
455     var token = createRawToken(type);
456     var preComments = List.copyOf(comments);
457     comments.clear();
458     readTrailingComments();
459     var postComments = List.copyOf(comments);
460     comments.clear();
461     return new Token(token, preComments, postComments);
462   }
463 
464   private CompilerException exception(String msg) {
465     Position pos = null;
466     if (eof) {
467       pos = new Position(filename, line, col);
468     } else {
469       pos = new Position(filename, startLine, startCol);
470     }
471     LOGGER.error(pos, msg);
472     LOGGER.print();
473     return new CompilerException("There were syntax errors");
474   }

476   private boolean isDigit() {
477     if (eof) {
478       return false;
479     }
480     return isDigit(input[index]);
481   }
482 
483   private boolean isDigit(byte c) {
484     return '0' <= c && c <= '9';
485   }
486 
487   private boolean isAlpha(byte c) {
488     return ('A' <= c && c <= 'Z') || ('a' <= c && c <= 'z') || c == '_';
489   }
490 
491   private boolean isAlphaNumeric() {
492     if (eof) {
493       return false;
494     }
495     return isAlphaNumeric(input[index]);
496   }
497 
498   private boolean isAlphaNumeric(byte c) {
499     return isDigit(c) || isAlpha(c);
500   }
