package org.mal_lang.lib;

public enum TokenType {
  SINGLECOMMENT("single line comment"),
  MULTICOMMENT("multi line comment"),
  EOF("end-of-file"),

  INCLUDE("'include'"),
  INFO("'info'"),
  CATEGORY("'category'"),
  ABSTRACT("'abstract'"),
  ASSET("'asset'"),
  EXTENDS("'extends'"),
  ASSOCIATIONS("'associations'"),
  LET("'let'"),
  EXIST("'E'"),
  C("'C'"),
  I("'I'"),
  A("'A'"),

  STRING("string literal"),
  ID("identifier"),
  INT("integer literal"),
  FLOAT("floating point literal"),
  HASH("'#'"),
  COLON("':'"),
  LCURLY("'{'"),
  RCURLY("'}'"),
  INHERIT("'+>'"),
  OVERRIDE("'->'"),
  ALL("'&'"),
  ANY("'|'"),
  NOTEXIST("'!E'"),
  AT("'@'"),
  LBRACKET("'['"),
  RBRACKET("']'"),
  LPAREN("'('"),
  RPAREN("')'"),
  COMMA("','"),
  REQUIRE("'<-'"),
  ASSIGN("'='"),
  UNION("'\\/'"),
  INTERSECT("'/\\'"),
  DOT("'.'"),
  RANGE("'..'"),
  STAR("'*'"),
  PLUS("'+'"),
  MINUS("'-'"),
  DIVIDE("'/'"),
  POWER("'^'"),
  LARROW("'<--'"),
  RARROW("'-->'");

  private final String string;

  private TokenType(String string) {
    this.string = string;
  }

  @Override
  public String toString() {
    return string;
  }
}
